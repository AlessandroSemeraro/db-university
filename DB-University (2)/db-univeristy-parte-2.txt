EX Query con Select

SELECT `date_of_birth` FROM `students` WHERE `date_of_birth` LIKE '1990%' ;
SELECT `cfu` FROM `courses` WHERE cfu > 10; 
SELECT * FROM `students` WHERE TIMESTAMPDIFF(YEAR, `date_of_birth`, CURDATE ()) > 30  
SELECT `period`,`year` FROM `courses` WHERE `period` = 'I semestre' AND `year` = '1'; 
SELECT `date`,`hour` FROM `exams` WHERE `date` = '2020/06/20' AND `hour` > '14:00'; 
SELECT `level` FROM `degrees` WHERE `level` = 'magistrale'; 
SELECT COUNT(*) FROM `departments`; 
SELECT `phone` FROM `teachers` WHERE `phone` IS NULL; 


EX Query con Group By

SELECT YEAR (`enrolment_data`) AS `anno_iscrizione` , COUNT(*)
FROM `students`
GROUP BY `anno_iscrizione`;

SELECT `office_address` , COUNT(*) 
FROM `teachers`
GROUP BY `office_address`;

SELECT `exam_id`, AVG (`vote`)
FROM `exam_student`
GROUP BY AVG(`exam_id`);

SELECT `department_id` , COUNT(*)
FROM `degrees`
GROUP BY `department_id`;
